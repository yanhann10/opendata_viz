Next we study only single-field academics and consolidate all academic field to one column, and filter for professors as others may have uneven level of skills and tool usage

survey=survey%>%
  filter(totalfield==1)

df2=survey%>%
  select(PHYS:ARTHUM)

whichfield=function(x) colnames(df2)[[which(df2[x,]==1)]]
df2$field=""
for (i in 1:nrow(df2)){
  df2$field[i]=whichfield(i)
}
df2=df2[-c(1:7)]
df3=cbind(survey,df2)
df3=df3[df3$ROLE=='Professor / Associate professor / Assistant professor',]

Remove the fields containing SPECCL which are free-text fields

occurence <- function(x) sum(as.numeric(x == unique(x[!is.na(x)])), na.rm=T)
toolkit=df3%>%
  select(field,GSCHOLSEARCH:BITBUCK)%>%
  group_by(field)%>%
  summarise_each(funs(occurence))%>%
  melt(id="field")
toolkit2=merge(toolkit,freqfield,by='field',all.x=T)
toolkit2$freq=toolkit2$value/toolkit2$num
toolkit2$value=NULL
toolkit2$num=NULL
toolkit2=toolkit2[!grepl("SPECCL", toolkit2$variable),]%>%
  arrange(freq)
